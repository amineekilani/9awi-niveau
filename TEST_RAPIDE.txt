VERIFICATION EN BASE :
----------------------

SELECT * FROM enrollments;
SELECT * FROM lecon_completions;

SELECT 
    u.email,
    c.titre,
    e.progress,
    COUNT(lc.id) as completed
FROM enrollments e
JOIN users u ON e.user_id = u.id
JOIN cours c ON e.cours_id = c.id
LEFT JOIN lecon_completions lc ON lc.enrollment_id = e.id
GROUP BY e.id;
